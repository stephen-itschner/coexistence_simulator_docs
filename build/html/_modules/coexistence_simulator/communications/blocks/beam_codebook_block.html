# communications/blocks/beam_codebook_block.py
"""Phased-array beam codebook helper.

A *codebook* is a JSON file that maps string indices to lists of complex
weights (I/Q pairs) or magnitude-phase tuples.  Example format:

```json
{
  "0": [[1, 0], [1, 0], [1, 0], [1, 0]],   // broadside (4 elems)
  "1": [[0.71, -0.71], [1, 0], [0.71, 0.71], [0, 1]],
  "2": [1, 1, 1, -1]                        // cartesian shorthand
}
```
* Each entry must resolve to a 1-D complex vector of identical length.
* Polar tuples `[mag, phase]` use *radians*.
* If `normalize=True`, every weight vector is divided by its L2-norm so
  that ||w||_2 = 1 (keeps Tx power constant across beams).

This block provides:
  * `process(index)`  -> weight vector `np.ndarray` (complex128).
  * `list_beams()`    -> sorted list of available beam indices.

It inherits from :class:`BaseBlock` so it can be chained like any other
low-level processing block.
"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Dict, List

import numpy as np

from coexistence_simulator.rf_system_blocks.blocks.base_block import BaseBlock

__all__ = ["BeamCodebookBlock"]

class BeamCodebookBlock(BaseBlock):
    """Phased-array beam codebook block.

    Loads a JSON codebook file at construction time and serves normalized
    complex weight vectors on demand.

    Args:
        codebook_path: File system path to the JSON codebook.
        normalize: If ``True``, each beam vector is divided by its L2 norm.

    Attributes:
        codebook_path: Path instance representing *codebook_path*.
        normalize: Whether normalization is enabled.
        num_elements: Number of antenna elements in each beam.
    """

    def __init__(self, codebook_path: str | Path, *, normalize: bool = True):
        """Create a :class:`BeamCodebookBlock`.

        Args:
            codebook_path: Path to the JSON codebook.
            normalize: When *True*, vectors are normalized to unit L2 norm.
        """

        super().__init__()
        self.codebook_path = Path(codebook_path)
        if not self.codebook_path.exists():
            raise FileNotFoundError(self.codebook_path)
        self.normalize = bool(normalize)
        self._load()

    # ------------------------------------------------------------------
    def _parse_entry(self, value) -> np.ndarray:
        """Convert a JSON entry into a complex weight vector.

        Args:
            value: Either a list of complex pairs or real magnitudes.

        Returns:
            Normalized complex NumPy array.

        Raises:
            ValueError: If *value* is not a one-dimensional Cartesian list or
                an ``Nx2`` polar list.
        """

        arr = np.asarray(value, dtype=float)

        # Case A: already 1-D real or complex list (cartesian)
        if arr.ndim == 1 and np.isrealobj(arr):
            # real values -> +/-1 phase-only weights
            comp = arr.astype(np.complex128)

        elif arr.ndim == 2 and arr.shape[1] == 2:
            # magnitude-phase pairs -> convert to cartesian
            mag = arr[:, 0]
            phase = arr[:, 1]
            comp = mag * np.exp(1j * phase)
        else:
            raise ValueError("Codebook entry must be 1-D cartesian or Nx2 polar list")

        if self.normalize:
            norm = np.linalg.norm(comp)
            if norm > 0:
                comp = comp / norm
        return comp

    def _load(self):
        """Read the codebook file into memory and validate its contents.

        Raises:
               ValueError: If the file contents are malformed or inconsistent.
        """

        data = json.loads(self.codebook_path.read_text())
        if not isinstance(data, dict):
            raise ValueError(
                "Codebook JSON must be an object mapping indices to vectors"
            )

        beams: Dict[int, np.ndarray] = {}
        for key, val in data.items():
            try:
                idx = int(key)
            except ValueError as e:
                raise ValueError(
                    f"Codebook keys must be integer strings (got {key!r})"
                ) from e
            beams[idx] = self._parse_entry(val)

        # sanity: all beams same length
        lengths = {len(v) for v in beams.values()}
        if len(lengths) != 1:
            raise ValueError("All beam weight vectors must have equal length")

        self._beams = beams
        self.num_elements = next(iter(lengths))

    def process(self, index: int) -> np.ndarray:
        """Return the complex weight vector for *index*.

        Args:
            index: Beam index.

        Returns:
            numpy.ndarray: Copy of the weight vector associated with *index*.

        Raises:
            ValueError: If *index* is not present in the codebook.
        """

        try:
            vec = self._beams[index]
        except KeyError as e:
            raise ValueError(f"Beam index {index} not found in codebook") from e
        return vec.copy()

    def list_beams(self) -> List[int]:
        """List available beam indices.

        Returns:
            list[int]: Sorted list of all beam indices in the codebook.
        """

        return sorted(self._beams)

    def get_parameters(self):
        """Serialize the block configuration.

        Returns:
            dict: Serializable description of the block configuration.
        """

        return {
            "block_name": self.block_name,
            "codebook_path": str(self.codebook_path),
            "normalize": self.normalize,
        }

    def set_parameters(self, parameters):
        """Load configuration values from *parameters*.

        Args:
            parameters: Dictionary as produced by
                :py:meth:`BeamCodebookBlock.get_parameters`.
        """

        self.block_name = parameters.get("block_name", self.block_name)
        path = parameters.get("codebook_path", str(self.codebook_path))
        normalize = parameters.get("normalize", self.normalize)
        if path != str(self.codebook_path) or bool(normalize) != self.normalize:
            self.codebook_path = Path(path)
            self.normalize = bool(normalize)
            self._load()
