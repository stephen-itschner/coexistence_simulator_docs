"""Zero‑stuff upsampler.

This block increases the sampling rate of a one‑dimensional NumPy array by an
integer factor *L* using the textbook zero‑stuffing method:

.. math::

    y[n] = \begin{cases}
        x[n / L], & n \equiv 0 \pmod{L} \\
        0,        & \text{otherwise}
    \end{cases}

No interpolation or filtering is applied; add a low‑pass filter afterward if
image suppression is required.
"""

from __future__ import annotations

import numpy as np

from coexistence_simulator.rf_system_blocks.blocks.base_block import BaseBlock

__all__ = ["Upsampler"]

class Upsampler(BaseBlock):
    """Insert zeros between samples to raise the sample rate.

    Args:
        factor: Positive integer upsampling factor. For example, ``factor=4``
            inserts three zeros after every input sample, producing an output
            that is four times longer than the input.

    Raises:
        ValueError: If *factor* is not an integer greater than or equal to 1.
    """

    def __init__(self, factor=2):
        super().__init__()
        self.factor = factor

    def process(self, input_signal):
        """Upsample ``input_signal`` by zero‑stuffing.

        Args:
            input_signal: One‑dimensional NumPy array (real or complex).

        Returns:
            NumPy array whose length equals ``len(input_signal) * factor``. The
            first element of the output equals ``input_signal[0]``; subsequent
            original samples appear every *factor*‑th position with zeros
            filling the gaps.
        """

        if len(input_signal) == 0:
            return np.array([])
        output_length = len(input_signal) * self.factor
        # Create a zero-filled array with the appropriate data type.
        output = np.zeros(output_length, dtype=input_signal.dtype)
        # Place the original samples at every 'factor'-th sample.
        output[:: self.factor] = input_signal
        return output
