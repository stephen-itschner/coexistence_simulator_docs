"""QPSK modulator and demodulator block.

The block converts binary data to QPSK constellation points and slices
received symbols back to bits. A single pad bit is added automatically
when the bit count is odd and removed on demodulation.
"""

import numpy as np

from coexistence_simulator.rf_system_blocks.blocks.base_block import BaseBlock

    def process(self, data, operation="modulate"):
        """Run QPSK modulation or demodulation.

        Args:
            data: One‑dimensional list or NumPy array that contains bits
                when ``operation`` is ``"modulate"`` or QPSK symbols when
                ``operation`` is ``"demodulate"``.
            operation: Either ``"modulate"`` (default) or
                ``"demodulate"``.

        Returns:
            np.ndarray: Complex QPSK symbols (modulation) or ``uint8``
            bits (demodulation).

        Raises:
            ValueError: If *operation* is not ``"modulate"`` or
                ``"demodulate"``.
        """

        if operation == "modulate":
            return self._modulate(data)
        elif operation == "demodulate":
            return self._demodulate(data)
        else:
            raise ValueError(
                "Invalid operation specified. Use 'modulate' or 'demodulate'."
            )



    def _modulate(self, data):
        """Map bits to QPSK symbols.

        Args:
            data: Iterable of bits (0 or 1). If the length is odd, a zero
                pad bit is appended automatically.

        Returns:
            np.ndarray: Complex QPSK symbols of length
            ``ceil(len(data) / 2)``.
        """

        data = np.array(data)
        self.padding_applied = False  # Reset padding flag

        # Padding the data if length is not a multiple of 2
        if len(data) % 2 != 0:
            data = np.append(data, 0)  # Pad with a 0 to make it even-length
            self.padding_applied = True  # Track that padding was applied

        # Reshape input bits to pairs and perform QPSK mapping
        reshaped_data = data.reshape(-1, 2)
        qpsk_signal = np.array(
            [1 - 2 * x[0] + 1j * (1 - 2 * x[1]) for x in reshaped_data]
        )

        return qpsk_signal

    def _demodulate(self, qpsk_signal):
        """Slice QPSK symbols into bits.

        Args:
            qpsk_signal: Iterable of complex QPSK symbols.

        Returns:
            np.ndarray: ``uint8`` bit array whose length equals
            ``2 * len(qpsk_signal)``, minus one if a pad bit was removed.
        """

        qpsk_signal = np.array(qpsk_signal)
        data_bits = []

        for symbol in qpsk_signal:
            bit_0 = 0 if symbol.real > 0 else 1
            bit_1 = 0 if symbol.imag > 0 else 1
            data_bits.extend([bit_0, bit_1])

        data_bits = np.array(data_bits)

        # Remove the padding bit if it was applied during modulation
        if self.padding_applied:
            data_bits = data_bits[:-1]  # Remove the last bit that was padded

        return data_bits
