class AGCBlock(BaseBlock):
    """Automatic Gain Control (AGC) block.

        The block rescales an input stream so that its average power converges to
        ``target_power``. A first‑order smoother with coefficient ``response_rate``
        updates the gain each call.
        """

    def __init__(self, target_power=1.0, response_rate=0.8):
        """Initialize the AGC block.

        Args:
            target_power: Desired average power after scaling. Defaults to
                ``1.0``.
            response_rate: Exponential smoothing factor in the open interval
                ``(0, 1]``. A value near ``1`` reacts quickly; a smaller value
                yields slower adaptation. Defaults to ``0.8``.
        """

        super().__init__()
        self.target_power = target_power
        self.response_rate = response_rate
        self.gain = None  # Initialize gain as None, to be set based on the first signal

    def process(self, input_signal):
        """Apply automatic gain control to *input_signal*.

        Args:
            input_signal: Real or complex samples as a one‑dimensional NumPy
                array.

        Returns:
            NumPy array with the same shape as *input_signal* whose mean power
            matches ``target_power``. If *input_signal* is empty the method
            returns an empty array. If the input power is zero a zero‑valued
            array is returned to avoid division by zero.
        """

        if len(input_signal) == 0:
            return np.array([])

        # Calculate current power of the input signal
        current_power = np.mean(np.abs(input_signal) ** 2)
        if current_power == 0:
            # Avoid division by zero
            return input_signal * 0.0

        # Calculate the gain needed to reach the target power
        desired_gain = np.sqrt(self.target_power / current_power)

        # If this is the first time running, set the gain directly to the desired value
        if self.gain is None:
            self.gain = desired_gain
        else:
            # Update the gain using the response rate (smooth adaptation)
            self.gain += self.response_rate * (desired_gain - self.gain)

        # Apply gain to the input signal
        output_signal = input_signal * self.gain

        return output_signal

    def get_parameters(self):
        """Return a serializable snapshot of the block state.

        Returns:
            Dictionary with keys ``block_name``, ``target_power``,
            ``response_rate``, and ``gain``. When the internal gain has not yet
            been set, the value ``"None"`` is used instead of ``None`` so the
            output is JSON‑friendly.
        """

        parameters = {
            "block_name": self.block_name,
            "target_power": self.target_power,
            "response_rate": self.response_rate,
            "gain": self.gain if self.gain is not None else "None",
        }
        return parameters

    def set_parameters(self, parameters):
        """Restore the block state from *parameters*.

        Args:
            parameters: Dictionary produced by :py:meth:`get_parameters`.
                Missing keys fall back to the constructor defaults.
        """

        self.target_power = parameters.get("target_power", 1.0)
        self.response_rate = parameters.get("response_rate", 0.8)
        self.gain = parameters.get("gain", None)
        if self.gain == "None":
            self.gain = None
