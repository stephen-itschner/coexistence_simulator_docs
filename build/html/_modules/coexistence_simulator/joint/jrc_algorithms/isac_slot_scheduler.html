# joint/jrc_algorithms/isac_slot_scheduler.py
"""ISAC slot scheduler that alternates communications frames with radar bursts.

The scheduler fronts two independent waveform blocks:

* **Communications block** – expects
  ``process(bits, operation="modulate")`` and returns a 1‑D complex
  baseband array.
* **Radar block** – expects either
  ``process(operation="modulate")`` *or* plain ``process()`` and returns a
  1‑D complex radar burst.

Policy
------
1. After each communications frame, the scheduler sends a radar burst.
2. The *cadence* parameter lets several successive communications frames
   occur before a radar burst is forced::

       cadence = 1  ➔  COMM, RADAR, COMM, RADAR, …
       cadence = 3  ➔  COMM×3, RADAR, COMM×3, RADAR, …

3. If no payload bits are queued, the scheduler always produces a radar
   burst, regardless of cadence.

The logic is intentionally lightweight for system‑level simulations and
does not enforce standard‑exact framing.
"""

from __future__ import annotations

from typing import List

import numpy as np

__all__ = ["ISACSlotScheduler"]

class ISACSlotScheduler:
    """Alternate communications slots with radar bursts according to cadence."""

    def __init__(self, comm_block, radar_block, *, cadence: int = 1):
        """Create a scheduler.

        Args:
            comm_block: Object exposing
                ``process(bits, operation="modulate")``. It must return a
                one‑dimensional complex NumPy array that represents a
                communications frame.
            radar_block: Object exposing either
                ``process(operation="modulate")`` or plain ``process()``.
                It must return a one‑dimensional complex NumPy array that
                represents a radar burst.
            cadence: Number of consecutive communications frames allowed
                before a radar burst is required. Must be **≥ 1**.

        Raises:
            ValueError: If *cadence* is less than one.
        """

        if cadence < 1:
            raise ValueError("cadence must be >= 1")
        self.comm_block = comm_block
        self.radar_block = radar_block
        self.cadence = cadence
        self._bit_queue: List[np.ndarray] = []
        self._comm_since_radar = 0
        self.last_action: str | None = None  # "comm" or "radar"

    def queue_bits(self, bits: np.ndarray):
        """Enqueue a communications payload.

        Args:
            bits: One‑dimensional ``uint8`` array of information bits. If
                *bits* is not already ``uint8``, it is cast before
                storage.
        """
        if bits.dtype != np.uint8:
            bits = bits.astype(np.uint8)
        self._bit_queue.append(bits)

    def queue_depth(self) -> int:
        """Return the number of queued communications payloads."""
        return len(self._bit_queue)

    def next_tx_frame(self) -> np.ndarray:
        """Return the next frame to transmit.

        The decision follows these rules:

        * If payload bits are queued **and** fewer than *cadence*
          communications frames have been sent since the last radar burst,
          send a communications frame.
        * Otherwise send a radar burst and reset the communications counter.

        Returns:
            One‑dimensional complex NumPy array containing either a
            communications frame or a radar burst.
        """

        # Decide whether to send comm or radar
        send_comm = bool(self._bit_queue) and self._comm_since_radar < self.cadence

        if send_comm:
            bits = self._bit_queue.pop(0)
            frame = self.comm_block.process(bits, operation="modulate")
            self._comm_since_radar += 1
            self.last_action = "comm"
            return frame
        else:
            # reset counter after radar burst
            self._comm_since_radar = 0
            try:
                frame = self.radar_block.process(operation="modulate")
            except TypeError:
                frame = self.radar_block.process()
            self.last_action = "radar"
            return frame
