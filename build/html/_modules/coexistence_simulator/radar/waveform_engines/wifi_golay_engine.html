"""Generate and detect Wi‑Fi Golay sequences.

The helper wraps :class:`GolaySequenceBlock` to provide:

* Cached A/B sequence generation for lengths 64, 128, or 256.
* Short‑training‑field (STF) construction for transmitters.
* FFT‑based matched filtering and a timing metric for receivers.
"""

from __future__ import annotations

import numpy as np

from coexistence_simulator.rf_system_blocks.blocks.sequence_block.golay_sequence_block import \
    GolaySequenceBlock

__all__ = ["WiFiGolayEngine"]

class WiFiGolayEngine:
    """Convenience wrapper for Golay sequences used in Wi‑Fi preambles.

    Args:
        length: Sequence length. Must be ``64``, ``128``, or ``256``.

    Raises:
        ValueError: If *length* is not one of the supported values.
    """

    def __init__(self, length: int = 128) -> None:
        """Create cached sequences of the given length."""
        if length not in (64, 128, 256):
            raise ValueError("length must be 64, 128 or 256")
        self.length = length
        self.a_seq, self.b_seq = GolaySequenceBlock.get_pair(length)

        # Cache FFTs for the matched filters (linear correlation via FFT)
        self._pad_len = 4 * length  # plenty for STF use-case
        self._h_a_fft = np.fft.fft(self.a_seq[::-1], self._pad_len)
        self._h_b_fft = np.fft.fft(self.b_seq[::-1], self._pad_len)

    def build_stf(self) -> np.ndarray:
        """Return the short‑training field (STF).

        The STF is ``A`` followed by ``B`` (concatenated) and cast to
        ``int8`` for compatibility with other preamble code.

        Returns:
            One‑dimensional NumPy array of signed bytes.
        """

        return np.concatenate((self.a_seq, self.b_seq)).astype(np.int8)



    def _corr(self, x: np.ndarray, seq: np.ndarray) -> np.ndarray:
        """Compute the circular correlation of *x* with *seq* via FFT.

        Args:
            x: Input complex sequence.
            seq: Reference Golay sequence (either ``A`` or ``B``).

        Returns:
            Complex array whose length equals ``len(x)``.
        """

        n_fft = max(self._pad_len, x.size + self.length - 1)
        x_fft = np.fft.fft(x.astype(np.complex64), n_fft)

        if n_fft != self._pad_len:
            h_fft = np.fft.fft(seq[::-1], n_fft)
        else:  # reuse cached 4*L FFT
            h_fft = self._h_a_fft if (seq is self.a_seq) else self._h_b_fft

        return np.fft.ifft(x_fft * h_fft, n_fft)[: x.size]

    def correlate(self, rx: np.ndarray) -> tuple[np.ndarray, np.ndarray]:
        """Correlate *rx* with the Wi‑Fi Golay A and B sequences.

        Args:
            rx: One‑dimensional complex baseband samples.

        Returns:
            Tuple ``(c_a, c_b)`` containing the two complex correlations.

        Raises:
            ValueError: If *rx* is not one‑dimensional.
        """

        if rx.ndim != 1:
            raise ValueError("rx must be 1-D")
        ca = self._corr(rx, self.a_seq)
        cb = self._corr(rx, self.b_seq)
        return ca, cb

    def timing_metric(self, rx: np.ndarray) -> np.ndarray:
        """Compute the STF timing metric for packet detection.

        Args:
            rx: One‑dimensional complex baseband samples.

        Returns:
            Real NumPy array of metric values, one per input sample.
        """

        ca, cb = self.correlate(rx)
        return np.abs(ca) ** 2 + np.abs(cb) ** 2
