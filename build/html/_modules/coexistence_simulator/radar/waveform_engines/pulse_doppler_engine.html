"""
Pulse‑Doppler burst engine.

The class synthesizes rectangular RF pulses at a fixed pulse‑repetition
frequency (PRF) and provides a matched‑filter receive path. Only NumPy is
required; no external DSP libraries are used.
"""

from __future__ import annotations

import numpy as np

__all__ = ["PulseDopplerEngine"]

class PulseDopplerEngine:
    """Lightweight rectangular‑pulse generator and matched filter.

    Args:
        pw: Pulse width in seconds.
        prf: Pulse‑repetition frequency in hertz.
        fs: ADC/DAC sample rate in hertz.

    Raises:
        ValueError: If any argument is non‑positive or if the pulse
            contains fewer than four samples.
    """

    def __init__(self, pw: float, prf: float, fs: float):
        if pw <= 0 or prf <= 0 or fs <= 0:
            raise ValueError("pw, prf and fs must be positive")
        self.pw = pw
        self.prf = prf
        self.fs = fs

        samples_per_pulse = int(round(pw * fs))
        if samples_per_pulse < 4:
            raise ValueError("pulse must contain at least 4 samples")
        self.n_samp_pulse = samples_per_pulse

        # Pre-compute one baseband rectangular pulse (complex 1+0j)
        self._tx_pulse = np.ones(self.n_samp_pulse, dtype=np.complex64)

        # Time between pulse starts, expressed in samples
        self.n_samp_cpi = int(round(fs / prf))

    def synth_burst(self, n_pulses: int) -> np.ndarray:
        """Generate a burst of **n_pulses** rectangular pulses.

        Args:
            n_pulses: Number of pulses to synthesize; must be positive.

        Returns:
            Complex NumPy array of shape ``(n_pulses, n_samp_cpi)`` where
            each row contains one pulse followed by zeros.

        Raises:
            ValueError: If *n_pulses* is not positive.
        """

        if n_pulses <= 0:
            raise ValueError("n_pulses must be positive")

        burst = np.zeros((n_pulses, self.n_samp_cpi), dtype=np.complex64)
        burst[:, : self.n_samp_pulse] = self._tx_pulse
        return burst

    def match_filter(self, rx: np.ndarray) -> np.ndarray:
        """Apply a matched filter to each received pulse.

        Args:
            rx: Complex NumPy array with shape
                ``(n_pulses, n_samp_cpi)``—one PRI per row.

        Returns:
            Complex NumPy array of the same shape as *rx* that contains
            the valid‑length convolution result for each pulse.

        Raises:
            ValueError: If *rx* is not two‑dimensional or the fast‑time
                dimension differs from ``n_samp_cpi``.
        """

        if rx.ndim != 2 or rx.shape[1] != self.n_samp_cpi:
            raise ValueError("rx must be 2-D with shape (n_pulses, n_samp_cpi)")
        filt = np.conjugate(self._tx_pulse[::-1])  # time-reversed
        # FFT convolution for efficiency on longer records
        n_fft = 1 << int(np.ceil(np.log2(self.n_samp_cpi + self.n_samp_pulse - 1)))
        pulse_fft = np.fft.fft(filt, n_fft)
        out = np.empty_like(rx, dtype=np.complex64)
        for k in range(rx.shape[0]):
            tmp = np.fft.ifft(np.fft.fft(rx[k], n_fft) * pulse_fft)[: self.n_samp_cpi]
            out[k] = tmp
        return out



    # ------------------------------------------ Helpers
    @property
    def pri_s(self) -> float:
        """Pulse-repetition interval in seconds."""
        return 1.0 / self.prf

    @property
    def duty_cycle(self) -> float:
        """Transmit duty cycle (pw / PRI)."""
        return self.pw * self.prf
