"""Cell‑Averaging CFAR detector for one‑dimensional magnitude vectors.

The helper contains no dependencies beyond NumPy, making it suitable for
unit tests and quick prototyping.

Example:
    ```python
    import numpy as np
    import matplotlib.pyplot as plt
    from coexistence_simulator.radar.signal_processing.cfar import ca_cfar_1d

    rng_prof = np.random.randn(256) * 0.5
    rng_prof[100] += 10  # Inject a strong target.

    det_mask, thresh = ca_cfar_1d(rng_prof, num_train=8,
                                  num_guard=2, pfa=1e-3)

    plt.plot(rng_prof)
    plt.plot(thresh, "--")
    plt.plot(np.where(det_mask)[0], rng_prof[det_mask], "rx")
    plt.show()
    ```
"""

from __future__ import annotations

import numpy as np

__all__ = ["ca_cfar_1d"]


def _calc_alpha(num_train: int, pfa: float) -> float:
    """Return the CA‑CFAR scaling factor.

    Args:
        num_train: Number of training cells on one side of the CUT
            (cell under test).
        pfa: Desired probability of false alarm, 0 < *pfa* < 1.

    Returns:
        Alpha, the threshold multiplier applied to the average noise
        power estimate.

    Raises:
        ValueError: If *num_train* is not positive or *pfa* is outside
            the open interval (0, 1).
    """

    if num_train <= 0:
        raise ValueError("num_train must be positive")
    if not (0 < pfa < 1):
        raise ValueError("pfa must be between 0 and 1")
    return num_train * (pfa ** (-1.0 / num_train) - 1.0)

def ca_cfar_1d(
    mag: np.ndarray,
    num_train: int = 8,
    num_guard: int = 2,
    pfa: float = 1e-3,
) -> tuple[np.ndarray, np.ndarray]:
    """Perform one‑dimensional Cell‑Averaging CFAR detection.

    Args:
        mag: One‑dimensional array of non‑negative magnitudes (for
            example, a range profile).
        num_train: Number of training cells on each side of the CUT.
        num_guard: Number of guard cells on each side of the CUT.
        pfa: Desired probability of false alarm (0 < pfa < 1).

    Returns:
        Tuple ``(det_mask, threshold)`` where

        * ``det_mask`` is a Boolean array marking detections.
        * ``threshold`` is an array of the same length that contains the
          calculated threshold at each index, or ``NaN`` where CFAR
          cannot be applied (edges).

    Raises:
        ValueError: If the input is empty, the CFAR window is larger than
            the signal, or any parameter is out of range.
    """

    if mag.size == 0:
        raise ValueError("Input magnitude vector is empty.")
    if pfa <= 0 or pfa >= 1:
        raise ValueError("pfa must be in (0,1)")
    if 2 * (num_train + num_guard) >= len(mag):
        raise ValueError("CFAR window larger than signal.")

    n = len(mag)
    det = np.zeros(n, dtype=bool)
    thr = np.full(n, np.nan)  # initialise to NaN  <- FIX

    alpha = num_train * (pfa ** (-1 / num_train) - 1)

    cut_lo = num_train + num_guard + 1
    cut_hi = n - cut_lo

    for idx in range(cut_lo, cut_hi):
        start_l = idx - num_guard - num_train
        end_l = idx - num_guard
        start_r = idx + num_guard + 1
        end_r = idx + num_guard + num_train + 1

        noise_level = (np.sum(mag[start_l:end_l]) + np.sum(mag[start_r:end_r])) / (
            2 * num_train
        )
        threshold = alpha * noise_level

        thr[idx] = threshold
        det[idx] = mag[idx] > threshold

    return det, thr
