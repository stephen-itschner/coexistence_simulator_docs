"""
Fast‑time (range) FFT helper.

The routine converts a stack of de‑chirped FMCW (or pulsed) returns
(``shape = (n_chirps, n_samples)``) into complex frequency spectra along
the fast‑time axis.

Key features
------------
* NumPy is the only dependency.
* Optional Hann windowing along fast time.
* Optional zero‑padding to the next power of two.

To map frequency bins to physical range, do the following outside this
function (the slope depends on the waveform):

    range_m = freq_hz * c0 / (2 * sweep_slope_hz_per_s)
"""

from __future__ import annotations

from typing import Tuple

import numpy as np

__all__ = ["range_fft"]

def range_fft(
    chirps: np.ndarray,
    fs: float,
    window: str = "hann",
    pad_pow2: bool = True,
) -> Tuple[np.ndarray, np.ndarray]:
    """Compute the range FFT along the fast‑time axis.

    Args:
        chirps: Complex or real NumPy array of shape
            ``(n_chirps, n_samples)``. Axis 1 is fast time.
        fs: ADC sample rate in hertz. Must be positive.
        window: Window applied along fast time before the FFT. Accepted
            values are ``"hann"`` and ``"none"``.
        pad_pow2: If ``True``, zero‑pad the fast‑time dimension to the
            next power‑of‑two length.

    Returns:
        Tuple ``(spec, freq_hz)`` where:

        * **spec** – Complex range spectrum with ``n_fft`` samples along
          axis 1 (fast time). The slow‑time dimension is unchanged.
        * **freq_hz** – One‑dimensional array of frequency bins in
          hertz, length ``n_fft`` (unshifted; 0 … +fs/2 … −fs/2).

    Raises:
        ValueError: If *chirps* is not 2‑D, *fs* is non‑positive, or
            *window* is invalid.
    """

    if chirps.ndim != 2:
        raise ValueError("chirps must be 2-D: (n_chirps, n_samples)")
    if fs <= 0:
        raise ValueError("fs must be positive")

    n_cpi, n_samp = chirps.shape
    n_fft = 1 << int(np.ceil(np.log2(n_samp))) if pad_pow2 else n_samp

    # Window along fast-time (axis-1)
    if window == "hann":
        win = np.hanning(n_samp)[None, :]  # row vector
    elif window == "none":
        win = 1.0
    else:
        raise ValueError("window must be 'hann' or 'none'")

    x = chirps * win

    # Zero-pad if requested
    if n_fft > n_samp:
        x = np.pad(x, ((0, 0), (0, n_fft - n_samp)))

    # FFT along fast-time axis
    spec = np.fft.fft(x, axis=1, n=n_fft)
    freq_hz = np.fft.fftfreq(n_fft, d=1.0 / fs)

    return spec, freq_hz
