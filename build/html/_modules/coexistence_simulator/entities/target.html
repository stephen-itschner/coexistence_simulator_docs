"""Simple radar target entity.

This module defines :class:`Target`, a stationary point scatterer with a
constant radar cross section (RCS).  The class inherits from
:class:`coexistence_simulator.entities.entity_base.EntityBase` so that it can
be used interchangeably with other simulation entities.
"""
from __future__ import annotations

from coexistence_simulator.entities.entity_base import EntityBase




[docs]
class Target(EntityBase):
    """Radar target with a fixed RCS.

    The target does not move or change orientation after construction.  Its
    radar cross section is represented by a single scalar value that stays
    constant for all aspect angles and frequencies.
    """

    def __init__(
        self,
        target_id: str,
        position: list[float],
        orientation: list[float],
        rcs: float,
    ) -> None:
        """Create a new target instance.

        Args:
            target_id: Unique string identifier for the target.
            position: Geographic coordinates ``[latitude, longitude, altitude]``.
                Latitude and longitude are in degrees; altitude is in meters.
            orientation: Body orientation ``[roll, pitch, yaw]`` in degrees.
            rcs: Radar cross section in square meters.
        """

        super().__init__(target_id, position, orientation)
        self.rcs = rcs  # Radar Cross Section in square meters
