"""
Delay–Doppler channel model.

The block produces an echo field that is the weighted sum of delayed and
Doppler‑shifted copies of the input complex‑baseband signal.

Implementation notes
--------------------
* Fractional delays are **realized** by linear interpolation.
* Doppler is applied as a complex phase ramp after the delay.
* Samples that reference time indices ``< 0`` or ``>= len(x)`` are filled
  with zero.

The design is intentionally lightweight so it can run inside unit tests
without heavy DSP dependencies.
"""

from __future__ import annotations

import numpy as np

from coexistence_simulator.rf_system_blocks.blocks.base_block import BaseBlock

class DelayDopplerChannelBlock(BaseBlock):
    """Lightweight delay–Doppler channel simulator.

    Args:
        fs: Sample rate in hertz. Must be positive.

    Raises:
        ValueError: If *fs* is not positive.
    """

    def __init__(self, fs: float):
        super().__init__()
        if fs <= 0:
            raise ValueError("fs must be positive")
        self.fs = float(fs)

    def process(
        self,
        x: np.ndarray,
        delays_s: np.ndarray,
        dopplers_hz: np.ndarray,
        gains: np.ndarray,
    ) -> np.ndarray:
        """Apply fractional delays and Doppler shifts to a complex signal.

        Args:
            x: Input complex‑baseband samples.
            delays_s: Per‑path delays in seconds.
            dopplers_hz: Per‑path Doppler shifts in hertz.
            gains: Complex path gains with one element per delay/Doppler
                pair.

        Returns:
            Complex NumPy array with the same shape and dtype as *x*.

        Raises:
            ValueError: If *delays_s*, *dopplers_hz*, and *gains* differ
                in length.
        """

        # --------------- argument checks -------------------------------
        if not (len(delays_s) == len(dopplers_hz) == len(gains)):
            raise ValueError("delays_s, dopplers_hz and gains must have equal length")

        n_samples = x.size
        t_vec = np.arange(n_samples) / self.fs

        # Real + imag interpolation tables
        idx_vec = np.arange(n_samples)
        x_r = np.real(x)
        x_i = np.imag(x)

        y = np.zeros_like(x, dtype=np.complex128)

        # ---------------- main loop ------------------------------------
        for tau, f_d, g in zip(delays_s, dopplers_hz, gains):
            # desired sample indices after delay
            frac_idx = idx_vec - tau * self.fs

            # Interpolate real and imag parts separately
            delayed_r = np.interp(frac_idx, idx_vec, x_r, left=0.0, right=0.0)
            delayed_i = np.interp(frac_idx, idx_vec, x_i, left=0.0, right=0.0)
            delayed = delayed_r + 1j * delayed_i

            # Apply Doppler phase rotation
            phase = np.exp(1j * 2.0 * np.pi * f_d * (t_vec - tau))
            path = g * delayed * phase

            y += path

        return y.astype(x.dtype, copy=False)
