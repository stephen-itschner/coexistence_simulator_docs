"""
Two‑ray ground‑reflection channel block.

The model sums two propagation paths:

1. Line‑of‑sight (LOS) path.
2. Single ground‑reflected path with complex reflection coefficient
   *Gamma*.

Free‑space loss is represented by 1/R amplitude scaling; the reflected
path is further multiplied by *Gamma*.

Geometry (side view)::

           h_t               h_r
    Tx o----|          |----o Rx
            |          |
            |          |
  -------------------------------- ground
                d

Distances
---------
R_los = sqrt(d**2 + (h_t − h_r)**2)
R_gnd = sqrt(d**2 + (h_t + h_r)**2)

Delays
------
tau_k = R_k / c,  where c ≈ 299 792 458 m/s

Gains
-----
g_los = 1 / R_los
g_gnd = Gamma / R_gnd
"""

from __future__ import annotations

import numpy as np

from coexistence_simulator.propagation_and_environment.channel_block.delay_doppler_channel_block import \
    DelayDopplerChannelBlock
from coexistence_simulator.rf_system_blocks.blocks.base_block import BaseBlock

class TwoRayGroundBlock(BaseBlock):
    """Two‑ray ground‑reflection channel simulator.

    Args:
        fs: Sample rate in hertz. Must be positive.
        reflection_coeff: Ground reflection coefficient *Gamma*. Default
            is ``-1+0j`` (perfectly conducting ground).

    Raises:
        ValueError: If *fs* is not positive.
    """

    C0 = 299_792_458.0  # speed of light (m/s)

    def __init__(self, fs: float, reflection_coeff: complex = -1.0 + 0j):
        super().__init__()
        if fs <= 0:
            raise ValueError("fs must be positive")
        self.fs = float(fs)
        self.reflection_coeff = complex(reflection_coeff)
        self._dd_chan = DelayDopplerChannelBlock(fs=fs)

    def process(
        self,
        x: np.ndarray,
        distance_m: float,
        tx_height_m: float,
        rx_height_m: float,
    ) -> np.ndarray:
        """Apply two‑ray fading to a complex waveform.

        Args:
            x: Input complex‑baseband samples.
            distance_m: Horizontal TX–RX spacing in meters.
            tx_height_m: Transmitter antenna height above ground in meters.
            rx_height_m: Receiver antenna height above ground in meters.

        Returns:
            Complex NumPy array of the same shape and dtype as *x* that
            contains LOS and ground‑reflected components.

        Raises:
            ValueError: If any geometry parameter is negative.
        """

        if distance_m < 0 or tx_height_m < 0 or rx_height_m < 0:
            raise ValueError("All geometry parameters must be non-negative")

        # Path lengths
        r_los = np.hypot(distance_m, tx_height_m - rx_height_m)
        r_gnd = np.hypot(distance_m, tx_height_m + rx_height_m)

        # Delays
        tau_los = round((r_los / self.C0) * self.fs) / self.fs
        tau_gnd = round((r_gnd / self.C0) * self.fs) / self.fs

        # Amplitude factors (1/R free-space loss)
        g_los = 1.0 / r_los
        g_gnd = self.reflection_coeff / r_gnd

        delays = np.array([tau_los, tau_gnd])
        dopplers = np.zeros_like(delays)
        gains = np.array([g_los, g_gnd], dtype=complex)

        return self._dd_chan.process(x, delays, dopplers, gains)
